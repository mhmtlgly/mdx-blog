## Create the Logo Component

```js
import React, { Fragment } from "react"
import { Link } from "gatsby"
import { makeStyles } from "@material-ui/core/styles"

const useStyles = makeStyles(theme => ({
  root: {
    borderBottom: props => props.brdr,
  },
}))

const SvgIcon = props => {
  const { Icon, height, width } = props
  const classes = useStyles(props)
  return (
    <Link className={classes.root} to="/">
      <Icon height={height} width={width} />
    </Link>
  )
}

export default SvgIcon
```

## Import and use the Logo Component

```js
import * as React from "react"
import Logo from "./logo"
import facebookIcon from "../../images/facebook-icon.svg"

export const Header = () => {
  return (
    <Fragment>
      <Logo Icon={facebookIcon} height={35} width={35} />
    </Fragment>
  )
}
```

## Install gatsby-plugin-react-svg

```js
// makes it possible to import SVG's as Components
    {
      resolve: "gatsby-plugin-react-svg",
      options: {
        rule: {
          include: /svg/,
        },
      },
    },
```
