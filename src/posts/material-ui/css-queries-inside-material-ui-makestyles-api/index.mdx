```js
const useStyles = makeStyles(theme => ({
  "@supports (display: grid)": {
    grid: {
      display: "grid",
      gridTemplateColumns: props =>
        `repeat(auto-fit, minmax(${props.gridwidth}px, 1fr))`,
      gridTemplateRows: "repeat(auto-fit, minmax(300px, 1fr))",
      gridGap: theme.spacing(2),
    },
  },
}))
```
